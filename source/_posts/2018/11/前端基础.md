手写数组去重
```js
function unique(arr){
    return [...new Set(arr)]
}
var arr = [1,1,'true','true',true,true,15,15,false,false, undefined,undefined, null,null, NaN, NaN,'NaN', 0, 0, 'a', 'a',{},{}];
console.log(unique(arr))
```
```js
function unique(arr){
   for (var i=0;i<arr.length;i++){
       for (var j=0;j<i<arr.length+1;j++){
           if(arr[i]===arr[j]){
               arr.splice(j,1);
                j--;
           }
       }
   }
}
var arr = [1,1,'true','true',true,true,15,15,false,false, undefined,undefined, null,null, NaN, NaN,'NaN', 0, 0, 'a', 'a',{},{}];
console.log(unique(arr))
```
